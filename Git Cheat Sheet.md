# Git Cheat Sheet - KZ Insight

# ----------------------------------------
# 1. معرفة حالة المشروع
# ----------------------------------------

git status

الوظيفة:
يعرض الملفات الجديدة، المعدلة، والمحذوفة.

استخدمه دائمًا قبل أي Commit.


# ----------------------------------------
# 2. إضافة الملفات إلى Staging Area
# ----------------------------------------

git add .

الوظيفة:
إضافة جميع الملفات الجديدة والمعدلة للتحضير للحفظ.

ملاحظة:
هذا لا يرفع الملفات إلى GitHub.


# ----------------------------------------
# 3. إضافة ملف واحد فقط
# ----------------------------------------

git add README.md

أو

git add src/main.py

الوظيفة:
إضافة ملف معين فقط.


# ----------------------------------------
# 4. إنشاء Commit
# ----------------------------------------

git commit -m "v0.1.0 - First Windows OS collector"

الوظيفة:
إنشاء Snapshot للمشروع داخل جهازك.

لا يتم الرفع إلى GitHub بعد.


# ----------------------------------------
# 5. رفع المشروع إلى GitHub
# ----------------------------------------

git push

الوظيفة:
رفع آخر Commit إلى GitHub.


# ----------------------------------------
# 6. جلب آخر التعديلات من GitHub
# ----------------------------------------

git pull

الوظيفة:
تحميل آخر نسخة من GitHub.


# ----------------------------------------
# 7. جلب آخر نسخة مع إعادة ترتيب التاريخ
# ----------------------------------------

git pull origin main --rebase

الوظيفة:
تحميل آخر نسخة من GitHub مع الحفاظ على تاريخ Commit نظيف.


# ----------------------------------------
# 8. مشاهدة سجل المشروع
# ----------------------------------------

git log

أو

git log --oneline

الوظيفة:
عرض جميع الـ Commits.


# ----------------------------------------
# 9. معرفة الفرع الحالي
# ----------------------------------------

git branch

الوظيفة:
عرض الفروع الموجودة وتحديد الفرع الحالي.


# ----------------------------------------
# 10. إنشاء فرع جديد
# ----------------------------------------

git branch feature/system-info

الوظيفة:
إنشاء فرع جديد.


# ----------------------------------------
# 11. الانتقال إلى فرع
# ----------------------------------------

git checkout feature/system-info

أو بالطريقة الحديثة

git switch feature/system-info


# ----------------------------------------
# 12. إنشاء والانتقال مباشرة
# ----------------------------------------

git checkout -b feature/system-info

أو

git switch -c feature/system-info


# ----------------------------------------
# 13. حذف ملف من Git
# ----------------------------------------

git rm filename


# ----------------------------------------
# 14. إعادة ملف قبل Commit
# ----------------------------------------

git restore filename


# ----------------------------------------
# 15. مشاهدة الفرق
# ----------------------------------------

git diff


# ----------------------------------------
# Workflow الصحيح
# ----------------------------------------

1)

git pull origin main --rebase

↓

2)

كتابة الكود

↓

3)

اختبار البرنامج

↓

4)

git status

↓

5)

git add .

↓

6)

git commit -m "رسالة واضحة"

↓

7)

git push


# ----------------------------------------
# أمثلة لرسائل Commit
# ----------------------------------------

v0.1.0 - Initial project structure

v0.2.0 - Windows operating system collector

v0.3.0 - CPU information collector

v0.4.0 - Memory information collector

v0.5.0 - Storage diagnostics

v0.6.0 - Network diagnostics

v0.7.0 - Report generator

v1.0.0 - First stable release

# لا تفعل

❌ git push قبل git pull إذا كان GitHub تغير.

❌ Commit لكل سطر كود.

# افعل دائمًا
# ----------------------------------------

✔ ارفعها إلى GitHub.
